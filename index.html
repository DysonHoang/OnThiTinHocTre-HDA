C++ · Template số học
// Kiểm tra số nguyên tố O(√N)
bool isPrime(long long n) {
    if (n < 2) return false;
    for (long long i = 2; i * i <= n; i++)
        if (n % i == 0) return false;
    return true;
}

// Sàng Eratosthenes
vector<bool> sieve(int N) {
    vector<bool> pr(N+1, true);
    pr[0] = pr[1] = false;
    for (int i = 2; i*i <= N; i++)
        if (pr[i])
            for (int j = i*i; j <= N; j += i) pr[j] = false;
    return pr;
}

// GCD, LCM
long long gcd(long long a, long long b){ return b ? gcd(b, a%b) : a; }
long long lcm(long long a, long long b){ return a / gcd(a,b) * b; }

// Lũy thừa nhanh (mod)
long long power(long long base, long long exp, long long mod) {
    long long res = 1;
    base %= mod;
    while (exp > 0) {
        if (exp & 1) res = res * base % mod;
        base = base * base % mod;
        exp >>= 1;
    }
    return res;
}

// Liệt kê tất cả ước của N
vector<int> divisors(int n) {
    vector<int> d;
    for (int i = 1; i*i <= n; i++)
        if (n%i==0) { d.push_back(i); if(i!=n/i) d.push_back(n/i); }
    sort(d.begin(), d.end());
    return d;
}

Chuyên đề 02

Tìm kiếm & Sắp xếpBinary Search — công cụ "giảm log" trong mọi bài toán tối ưu

Tìm kiếm nhị phân — khuôn mẫu chuẩn

Binary search không chỉ để tìm phần tử trong mảng — nó là công cụ tối ưu hóa: "Binary search on answer" — tìm giá trị nhỏ nhất / lớn nhất thỏa mãn điều kiện.

Tìm vị trí trong mảng

Mảng sắp xếp tăng dần. Tìm vị trí đầu tiên có a[mid] ≥ target (lower_bound) hoặc a[mid] > target (upper_bound).

Binary search on answer

Khi bài hỏi "tìm giá trị nhỏ nhất X sao cho f(X) = true" — binary search trên X, kiểm tra điều kiện trong O(N).

C++ · Binary Search Templates
// Template chuẩn — lower_bound tự viết
int lower_bound(vector<int>& a, int x) {
    int lo = 0, hi = a.size();
    while (lo < hi) {
        int mid = (lo + hi) / 2;
        if (a[mid] < x) lo = mid + 1;
        else            hi = mid;
    }
    return lo;
}

// Binary search on answer: tìm số ngày ít nhất để sản xuất đủ hàng
bool check(int days, vector<int>& rate, long long need) {
    long long total = 0;
    for (int r : rate) total += (long long)r * days;
    return total >= need;
}

int lo = 1, hi = 1e9, ans = hi;
while (lo <= hi) {
    int mid = (lo + hi) / 2;
    if (check(mid, rate, need)) { ans = mid; hi = mid - 1; }
    else lo = mid + 1;
}

// Prefix sum 2D
int ps[1001][1001];
for(int i=1;i<=n;i++)
  for(int j=1;j<=m;j++)
    ps[i][j] = a[i][j] + ps[i-1][j] + ps[i][j-1] - ps[i-1][j-1];

// Truy vấn tổng hình chữ nhật (r1,c1)-(r2,c2)
auto query = [&](int r1,int c1,int r2,int c2) {
    return ps[r2][c2]-ps[r1-1][c2]-ps[r2][c1-1]+ps[r1-1][c1-1];
};

// Two pointers: tìm đoạn con có tổng đúng bằng S
int l=0, sum=0;
for(int r=0; rwhile(sum > S) sum -= a[l++];
    if(sum == S) ans++;
}

// Sliding window max — dùng deque

for(int i=0; iwhile(!dq.empty() && a[dq.back()] <= a[i]) dq.pop_back();
    dq.push_back(i);

Chuyên đề 04

Quy hoạch độngDynamic Programming — trọng tâm của mọi kỳ thi tin học

Nhận diện bài toán DP

Bài toán có thể giải bằng DP khi thỏa hai tính chất:

  • Tính chất tối ưu con: lời giải tối ưu của bài toán chứa lời giải tối ưu của các bài toán con.
  • Bài toán con trùng lặp: các bài toán con giống nhau được giải nhiều lần nếu dùng đệ quy thông thường.
Quy trình 4 bước

1. Định nghĩa trạng thái dp[...]  2. Viết công thức chuyển  3. Khởi tạo base case  4. Xác định thứ tự tính

Các dạng cổ điển

LIS · LCS · Ba lô 0/1 · Đổi tiền · Cắt que · Tam giác · DP trên cây · DP trên interval

C++ · LIS O(N log N)
// Dãy con tăng dài nhất — O(N log N) với patience sorting
int LIS(vector<int>& a) {
    vector<int> dp; // dp[i] = phần tử nhỏ nhất kết thúc dãy tăng độ dài i+1
    for (int x : a) {
        auto it = lower_bound(dp.begin(), dp.end(), x);
        if (it == dp.end()) dp.push_back(x);
        else *it = x;
    }
    return dp.size();
}

// Ba lô 0/1 — N vật, trọng lượng tối đa W
vector<int> dp(W+1, 0);
for (int i = 0; i < N; i++)
    for (int w = W; w >= weight[i]; w--)
        dp[w] = max(dp[w], dp[w - weight[i]] + value[i]);

// LCS — Dãy con chung dài nhất
for (int i = 1; i <= n; i++)
    for (int j = 1; j <= m; j++)
        if (a[i] == b[j]) dp[i][j] = dp[i-1][j-1] + 1;
        else             dp[i][j] = max(dp[i-1][j], dp[i][j-1]);

Tham lam đúng khi thỏa Greedy Choice Property: lựa chọn tốt nhất tại mỗi bước luôn dẫn đến lời giải tổng thể tối ưu. Hai cách chứng minh phổ biến:

// Dijkstra với priority_queue
vector<long long> dijkstra(int src, vector<vector<pair<int,int>>>& adj) {
    int n = adj.size();
    vector<long long> dist(n, LLONG_MAX);
    priority_queue<pair<long long,int>, vector<pair<long long,int>>, greater<>> pq;
    dist[src] = 0;
    pq.push({0, src});
    while (!pq.empty()) {
        auto [d, u] = pq.top(); pq.pop();
        if (d > dist[u]) continue;
        for (auto [v, w] : adj[u])
            if (dist[u] + w < dist[v])
                pq.push({dist[v] = dist[u] + w, v});
    }
    return dist;
}

// DSU (Disjoint Set Union) cho Kruskal
struct DSU {
    vector<int> p, rank_;
    DSU(int n) : p(n), rank_(n, 0) { iota(p.begin(), p.end(), 0); }
    int find(int x) { return p[x] == x ? x : p[x] = find(p[x]); }
    bool unite(int a, int b) {
        a = find(a); b = find(b);
        if (a == b) return false;
        if (rank_[a] < rank_[b]) swap(a, b);
        p[b] = a; if(rank_[a]==rank_[b]) rank_[a]++;
        return true;
    }
};

// Fenwick Tree (Binary Indexed Tree)
struct BIT {
    vector<int> tree;
    int n;
    BIT(int n) : n(n), tree(n+1, 0) {}
    void update(int i, int v){ for(;i<=n;i+=i&-i) tree[i]+=v; }
    int  query(int i){ int s=0; for(;i>0;i-=i&-i) s+=tree[i]; return s; }
    int  query(int l, int r){ return query(r)-query(l-1); }
};

// Segment Tree (Sum, lazy add)
struct SegTree {
    int n;
    vector<long long> tree, lazy;
    SegTree(int n) : n(n), tree(4*n,0), lazy(4*n,0) {}
    void push(int node, int l, int r) {
        if(lazy[node]){
            int m=(l+r)/2;
            tree[2*node] += lazy[node]*(m-l+1);
            tree[2*node+1] += lazy[node]*(r-m);
            lazy[2*node] += lazy[node];
            lazy[2*node+1] += lazy[node];
            lazy[node] = 0;
        }
    }
    void update(int node,int l,int r,int ql,int qr,long long v){
        if(qrreturn;
        if(ql<=l&&r<=qr){ tree[node]+=v*(r-l+1); lazy[node]+=v; return; }
        push(node,l,r); int m=(l+r)/2;
        update(2*node,l,m,ql,qr,v); update(2*node+1,m+1,r,ql,qr,v);
        tree[node]=tree[2*node]+tree[2*node+1];
    }
};
